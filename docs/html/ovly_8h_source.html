66static inline void ovly_load_text(uint32_t index, uint32_t chan) {
67 uint32_t lma = ovly_table[index].lma; // Load address in AXI space.
68 uint32_t rma = ovly_table[index].rma; // Run address in VCPU PMEM space.
69 size_t size = ovly_table[index].size;
70 dmac_enable(DMAC_VPRAM | chan, size, lma, rma);
71}
