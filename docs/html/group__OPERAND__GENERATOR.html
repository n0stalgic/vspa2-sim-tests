          (
           
          __val)
          
        

      



Value:
    do { \


        if (__builtin_isintconst(__val)) \


        { __asm("mv $nco_k, %0" : : "i"(__val)); } \


        else \


        { __asm("mv $nco_k, %0" : : "x"(__val)); } \


    } while (0)


Prototype:
void __set_nco_k(Type val)


Description:
Initialize the k-parameter of the NCO


Initialize the k-parameter of the NCO to val. Note that nco_k is a half
word register and hence only the least significant portion of val is
used. In case of constant, this only affects the least significant
10 bits of the half-word nco_k register.
Type can be const int, int, unsigned int, float

Underlying assembly:
mv NCO_REG,GP mv nco_k,NCO_K


Support:
VSPA
