22// Trigger options:
23#define DMAC_TRIG_IPPU (0b1 << 12)
24#define DMAC_TRIG_VCPU (0b1 << 13)
25#define DMAC_TRIG_IRQ (0b1 << 14)
26#define DMAC_TRIG_FECU (0b1 << 22)
27
28// Pending options:
29#define DMAC_PEND_EXT (0b1 << 16)
30#define DMAC_PEND_IPPU (0b1 << 17)
31#define DMAC_PEND_FECU (0b1 << 23)
32
33// Pointer reset request:
34#define DMAC_PRST_REQ (0b1 << 15)
35
36// FIFO mode:
37#define DMAC_FIFO_HS ((0b1 << 11))
38#define DMAC_FIFO ((0b1 << 11) | DMAC_PEND_EXT)
39#define DMAC_FIFO_RESET ((0b1 << 11) | DMAC_PRST_REQ)
40
41// Program memory access:
42#define DMAC_VPRAM (0b010 << 8)
43#define DMAC_IPRAM (0b011 << 8)
44
45// Turbo read:
46#define DMAC_MBRE (0b1 << 7)
47
48// Transfer modes:
49#define DMAC_RD (0b000 << 8)
50#define DMAC_RDC (0b001 << 8)
51#define DMAC_WR (0b110 << 8)
52#define DMAC_WRC (0b111 << 8)
53#define DMAC_DI8 ((0b100 << 8) | (0b001 << 18))
54#define DMAC_DI4_MSB0 ((0b100 << 8) | (0b010 << 18))
55#define DMAC_DI4_MSB1 ((0b100 << 8) | (0b011 << 18))
56#define DMAC_DI16 ((0b100 << 8) | (0b100 << 18))
57#define DMAC_DI32 ((0b100 << 8) | (0b101 << 18))
